    GQL 语法汇总¶

    本文盘点了 GQL 中最常用的语句与命令，方便快速查阅。

    Note

    本文中的语句与命令主要基于电影数据集。为了简洁起见，在多数数据查询语句中省略了 USE movie 子句。

    数据定义语句¶

    Schema¶

    语句 描述 
    CREATE SCHEMA IF NOT EXISTS /movie
    		 创建 Schema。 
    DROP SCHEMA IF EXISTS /movie
    		 销毁 Schema。

    图类型¶

    语句 描述 
    '''
CREATE GRAPH TYPE IF NOT EXISTS movie_type AS { 
    NODE Actor (LABEL Person {id INT64, name STRING, birthDate DATE, PRIMARY KEY (id)}) , 
    NODE Director (LABEL Person {id INT64, name STRING, birthDate DATE, PRIMARY KEY (id)}), 
    NODE `User` (LABEL Person {id INT64 PRIMARY KEY}), 
    NODE Movie (LABEL Movie {id INT64 PRIMARY KEY, name STRING}), 
    NODE Genre (LABEL Genre {id INT64 PRIMARY KEY, name STRING}), 
    EDGE Act (Actor)-[:Act]->(Movie), 
    EDGE Direct (Director)-[:Direct]->(Movie), 
    EDGE Watch (`User`)-[:Watch {rate FLOAT}]->(Movie), 
    EDGE WithGenre (Movie)-[:WithGenre]->(Genre) }
'''
    		创建每个点类型均使用一个属性作为主键的图类型。 
    ''' 
CREATE GRAPH TYPE IF NOT EXISTS social_network AS { 
    NODE Person (LABEL Person {id INT64 PRIMARY KEY, name STRING}), 
    NODE Company (LABEL Company {id INT64 PRIMARY KEY, name STRING}), 
    EDGE WorkAt (Person)-[:WorkAt {`from` DATE, to DATE, MULTIEDGE KEY ()}]->(Company), 
    EDGE Friend (Person)-[:Friend {since DATE MULTIEDGE KEY }]->(Person) }
'''
    		创建边类型均使用一个属性作为多边键的图类型。 
    '''
CREATE GRAPH TYPE IF NOT EXISTS social_network_2 AS { 
    NODE Person (LABEL Person {id INT64 PRIMARY KEY, name STRING DEFAULT 'unknown', age INT32 DEFAULT 0}), 
    NODE Company (LABEL Company {id INT64 PRIMARY KEY, name STRING}) }
'''
    		创建点类型的属性带有默认值的图类型。 
    '''
CREATE GRAPH TYPE IF NOT EXISTS primary_key_multi_properties AS { 
    NODE Person (LABEL Person {id INT64, name STRING, PRIMARY KEY (id, name)}), 
    EDGE Know (Person)-[:`Like` {`from` DATE, to DATE, MULTIEDGE KEY (`from`,to)}]->(Person) }
'''
    		创建主键和多边键分别由多个属性组成的图类型。 
    CREATE GRAPH TYPE IF NOT EXISTS empty_graph_type AS {}
    		 创建空图类型。 
    DROP GRAPH TYPE movie_type
    		 销毁图类型。 
    '''
ALTER GRAPH TYPE movie_type { 
    ADD NODE TYPE Company (LABEL Company {id INT PRIMARY KEY, name STRING}), 
    ADD EDGE TYPE Serve (Director)-[LABEL Serve{start_year INT, end_year INT}]->(Company) }
'''
    		添加点类型和边类型。
    ''' ALTER GRAPH TYPE movie_type { DROP NODE TYPE Company, DROP EDGE TYPE Serve } ''' 删除点类型和边类型。 
    '''
ALTER GRAPH TYPE movie_type { 
    RENAME NODE TYPE Actor TO Actor_renamed, 
    RENAME EDGE TYPE Act TO Act_renamed }
'''
    		重命名点类型和边类型。 
    '''
ALTER GRAPH TYPE movie_type { 
    ALTER NODE TYPE `User` ADD LABELS Male&Above30, 
    ALTER EDGE TYPE Watch ADD LABELS InCinema&AtHome }
'''
    		为点类型和边类型添加标签。 
    '''
ALTER GRAPH TYPE movie_type { 
    ALTER NODE TYPE `User` DROP LABELS Male&Above30, 
    ALTER EDGE TYPE Watch DROP LABELS InCinema&AtHome }
'''
    		为点类型和边类型删除标签。 
    '''
ALTER GRAPH TYPE movie_type { 
    ALTER NODE TYPE `User` ADD PROPERTIES {email STRING, age INT64}, 
    ALTER EDGE TYPE Watch ADD PROPERTIES {`count` INT64, platform STRING} }
'''
    		为点类型和边类型添加属性。 
    '''
ALTER GRAPH TYPE movie_type { 
    ALTER NODE TYPE `User` DROP PROPERTIES {email, age}, 
    ALTER EDGE TYPE Watch DROP PROPERTIES {`count`, platform} }
'''
    		为点类型和边类型删除属性。 
    '''
ALTER GRAPH TYPE movie_type { 
    ALTER NODE TYPE Director RENAME PROPERTIES {birthDate TO birth, name TO lastName }, 
    ALTER EDGE TYPE Watch RENAME PROPERTIES {`rate` TO rating} }
'''
    		重命名点类型和边类型的属性。 
    '''      
ALTER GRAPH TYPE movie_type { 
    ALTER NODE TYPE Director MODIFY PROPERTIES {name STRING DEFAULT "Alex", birthDate DATE DEFAULT date("2016-09-20") },
    ALTER EDGE TYPE Watch MODIFY PROPERTIES {rate FLOAT DEFAULT 4.5} }
'''
    		修改点类型和边类型的属性。 
    RENAME GRAPH TYPE movie_type TO movie_type_renamed
    		 重命名图类型。

    图¶

    语句 描述 
    CREATE GRAPH IF NOT EXISTS movie TYPED movie_type
    		 创建图。 
    DROP GRAPH movie
    		 销毁图。 
    RENAME GRAPH movie TO movie_renamed
    		 重命名图。

    临时图¶

    语句 描述 
    CREATE TEMPORARY GRAPH IF NOT EXISTS #movie_temporary_graph AS COPY OF movie
    		 基于现有图创建临时图。 
    CREATE TEMPORARY GRAPH IF NOT EXISTS #movie_temporary_graph_3 AS COPY OF GRAPH { USE movie MATCH (v) RETURN v }
    		 基于子图创建临时图。 
    DROP TEMPORARY GRAPH IF EXISTS #movie_temporary_graph
    		 销毁临时图。

    索引¶

    语句 描述 
    USE movie CREATE NORMAL INDEX IF NOT EXISTS idx_actor_age ON NODE Actor ( birthDate ASC )
    		 在点类型上创建普通索引。 
    USE movie CREATE NORMAL INDEX IF NOT EXISTS idx_watch_rating ON EDGE Watch ( rate DESC )
    		 在边类型上创建普通索引。 
    '''
USE ann_test_graph CREATE VECTOR INDEX ann_node_ivf_l2 ON NODE N1&N2::vec1 
      OPTIONS {DIM: 3, METRIC: L2, TYPE: IVF, NLIST: 10, TRAINSIZE: 4 }
'''
    		基于度量L2，在两个点类型上创建 IVF 向量索引。 
    '''
USE ann_test_graph CREATE VECTOR INDEX ann_edge_hnsw_ip ON EDGE E1::vec1 
      OPTIONS {DIM: 3, METRIC: IP, TYPE: HNSW, MAXDEGREE: 3, EFCONSTRUCTION: 6, CAPACITY: 500
'''
    		基于度量IP，在一个边类型上创建 HNSW 向量索引。 
    USE my_graph CREATE SPATIAL INDEX IF NOT EXISTS idx_gnode_g ON NODE Person(any_geo, s2_max_level = 30, s2_max_cells = 8)
    		 在点类型上基于地理空间类型的属性创建空间索引。 
    USE my_graph CREATE SPATIAL INDEX IF NOT EXISTS idx_gedge_g ON EDGE GEdge(company_location, s2_max_level = 30, s2_max_cells = 8)
    		 在边类型上基于地理空间类型的属性创建空间索引。 
    USE movie CREATE FULLTEXT INDEX IF NOT EXISTS ft_movie ON NODE Movie(name)

    匹配计算语句¶

    语句 描述 
    '''
USE #match_compute_movie {
  /* ====== 声明 ====== */
  VALUE source_id = 1                          // 用于相似度计算的源用户 ID
  VALUE source_edges    SumAgg = 0      // 源用户喜欢的电影
  VALUE topk_similarity TopKAgg<5, similarity DOUBLE DESC, id INT DESC>  // 前 5 位相似用户
  NODE VALUE target_edges SumAgg = 0    // 每个用户喜欢的电影
  NODE VALUE intersection SumAgg = 0    // 与源用户的共同喜欢的电影
  VALUE v_set ACTIVE_SET                       // 源用户喜欢的电影

  /* ====== 步骤 1：统计用户电影偏好 ====== */
  // 统计所有用户喜欢的电影
  MATCH (u@User)-[e:Watch]->(m:Movie)
  PER PATH {
    SET u.@target_edges += 1                  // 每个用户的电影计数递增
  }

  /* ====== 步骤 2：收集源用户数据 ====== */
  // 处理源用户的偏好
  MATCH (u@User)-[:Watch]->(m:Movie) 
  WHERE u.id = source_id  // 筛选源用户
  PER NODE (u) {
    SET @source_edges += u.@target_edges     // 存储源用户的电影计数
  }
  FINALLY { SET v_set = m }                      // 用源用户的电影填充活动集

  /* ====== 步骤 3：计算用户相似度 ====== */
  // 查找源用户与其他用户的共同喜欢电影
  MATCH (m:Movie)<-[:Watch]-(t@User) 
  WHERE t.id <> source_id AND m IN v_set                  // 排除源用户，使用源用户的电影作为起点
  PER PATH {
    SET t.@intersection += 1                 // 统计共同电影
  }
  PER NODE (t) {
    // 计算源用户和其他用户的共同喜欢电影的并集
    VALUE union_size = t.@target_edges + @source_edges - t.@intersection

    IF union_size <> 0 THEN {
      VALUE similarity = CAST(t.@intersection AS DOUBLE) / union_size

      // 存储到前 k 聚合器
      SET @topk_similarity += RECORD{
        similarity: similarity, 
        id: t.id
      }
    }
  }

  /* ====== 步骤 4：返回结果 ====== */
  // 输出前 5 位相似用户
  FOR r IN @topk_similarity
    RETURN r.id AS user_id, 
          ROUND(r.similarity, 5) AS jaccard_similarity
}
'''
    		基于用户的电影偏好计算用户之间的 Jaccard 相似度。

    会话¶

    命令 描述 
    SESSION SET timezone = "America/New_York"
    		 设置会话配置项。 
    SESSION SET VALUE $i_1 = 10, $i_2 = 20
    		 为当前会话设置参数。 
    SESSION UNSET timezone
    		 取消当前会话中的时区设置。 
    SESSION RESET timezone
    		 重置会话配置项。 
    SESSION RESET $i_1, $i_2
    		 删除指定会话参数。 
    SESSION CLOSE
    		 关闭当前会话。 
    SHOW CURRENT_SESSION
    		 显示当前用户创建的会话。 
    SHOW CURRENT_SESSION TIME FORMATS
    		 显示当前会话中的日期和时间格式。 
    SHOW SESSIONS
    		 显示当前服务组中 Graph 服务上所有活跃的会话。
